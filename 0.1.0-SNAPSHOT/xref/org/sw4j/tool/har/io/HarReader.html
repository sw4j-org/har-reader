1   package org.sw4j.tool.har.io;
2   
3   import com.google.gson.Gson;
4   import com.google.gson.GsonBuilder;
5   import java.io.Reader;
6   import org.sw4j.tool.har.model.Har;
7   import org.sw4j.tool.har.model.Log;
8   
9   /**
10   * <p>
11   * A reader that reads HAR JSON from a reader and returns the parsed model.
12   * </p>
13   *
14   * @author Uwe Plonus &lt;u.plonus@gmail.com&gt;
15   */
16  public class HarReader {
17  
18      /** The source of the HAR JSON model. */
19      private final Reader source;
20  
21      /**
22       * <p>
23       * A constructor that takes a reader for this HAR reader.
24       * </p>
25       *
26       * @param reader the reader that will be used for parsing.
27       */
28      public HarReader(final Reader reader) {
29          this.source = reader;
30      }
31  
32      /**
33       * <p>
34       * Reads in the JSON from the reader and returns a model from the parsed JSON.
35       * </p>
36       *
37       * @param check whether to check the JSON for all required attributes.
38       * @return the log that is parsed from the JSON.
39       * @throws AttributeRequiredException if {@code check} is set and a required attribute is not set.
40       */
41      public Log read(final boolean check) throws AttributeRequiredException {
42          Gson gson = new GsonBuilder().excludeFieldsWithoutExposeAnnotation().create();
43          Har har = gson.fromJson(source, Har.class);
44          if (check) {
45              HarValidator.checkRequiredAttributes(har);
46          }
47          return har.getLog();
48      }
49  
50  }
