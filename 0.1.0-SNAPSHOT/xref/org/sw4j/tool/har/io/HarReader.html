1   package org.sw4j.tool.har.io;
2   
3   import com.google.gson.Gson;
4   import com.google.gson.GsonBuilder;
5   import java.io.Reader;
6   import org.sw4j.tool.har.model.Har;
7   import org.sw4j.tool.har.model.Log;
8   
9   public class HarReader {
10  
11      private final Reader source;
12  
13      public HarReader(Reader reader) {
14          this.source = reader;
15      }
16  
17      public Log read(boolean check) {
18          Gson gson = new GsonBuilder().excludeFieldsWithoutExposeAnnotation().create();
19          Har har = gson.fromJson(source, Har.class);
20          if (check) {
21              HarValidator.checkRequiredAttributes(har);
22          }
23          return har.getLog();
24      }
25  
26  }
